Наблюдатель/Observer

Наблюдатель это поведенческий паттерн, который позволяет одним объектам (Наблюдателям) реагировать на изменения происходящие в других объектах (Издателях)

Псевдокод класса Наблюдателя

interface IObserver {
    update(param);
}

class Observer implements IObserver {
    update(param) {
        //Логика реагирования на возникшее событие
    }
    //Логика необходимая классу по его назначению
}
Псевдокод класса Издателя

interface ISubject {
    subscribe(observer: IObserver);
    unsubscribe(observer: IObserver);
    notify();
}

class Subject implements ISubject {
    private observers = [];
    private parameter = '';

    subscribe(observer: IObserver) {
        observers.add(observer);
    }
    unsubscribe(observer: IObserver) {
        observers.remove(observer);
    }
    notify() {
        observers.forEach((observer) => observer.update(parameter));
    }
    //Логика необходимая классу по его назначению
    sendEvent() {
        this.notify();
    }
}
Псевдокод использования класса

const subject = new Subject();

const someObserver = new Observer();
const anotherObserver = new Observer();

subject.subscribe(someObserver);
subject.subscribe(anotherObserver);

subject.senEvent();
subject.notify();
Одиночка/Singleton

Одиночка это порождающий паттерн, который обеспечит существование только одного экземпляра класса

Псевдокод класса

class Singleton {
    private static instance = new Singleton();
    private constructor() {}

    static getInstance() {
        return Singleton.instance;
    }
}
Псевдокод использования класса

const first = Singleton.getInstance();
const second = Singleton.getInstance();
first === second //true
