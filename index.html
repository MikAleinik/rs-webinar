Наблюдатель/Observer

Наблюдатель это поведенческий паттерн, который позволяет одним объектам реагировать на изменения происходящие в других объектах

Псевдокод класса

interface IListener {
    send();
}

class Observer {
    subscribe(listener: IListener) {
        listeners.add(listener);
    }
    unsubscribe(listener) {
        listeners.remove(listener);
    }
    notify() {
        listeners.forEach((listener) => listener.send());
    }
}
